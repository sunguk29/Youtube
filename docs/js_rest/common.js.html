/**
 * LoginSubmit,
 * 로그인폼 validation 함수
 * @requires [inspection]
 * @return {boolean}
 * */
const loginSubmit = () => {
  return inspection({
    selector: '[name="email"]',
    regex_type: 'email',
    empty_text: '이메일을 입력해주세요.',
    failed_text: '이메일을 정확히 입력해주세요.',
  }) && inspection({
    selector: '[name="password"]',
    regex_type: 'password',
    empty_text: '패스워드를 입력해주세요.',
    failed_text: '패스워드를 정확히 입력해주세요. (8자 이상 20자 이하, 영문, 숫자, 특수문자(@$!%*#?&^) 모두 포함)',
  });
};
